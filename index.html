🔧 Google Apps Script Setup Instructions

Password is now stored securely in Apps Script — not in this HTML file. Follow these steps to set it up.

  1. Go to your Google Sheet → Extensions → Apps Script.
  2. Replace the entire script with the code below and save it.
  3. On line 2, change your-strong-password-here to your actual password.
  4. Click Deploy → Manage Deployments. Click the pencil icon, change version to "New version", and click Deploy.
  5. Ensure "Execute as: Me" and "Who has access: Anyone" are selected.
  6. ✅ Your password is now only in Apps Script — never visible in public HTML code.
// ── STEP 1: Set your password here (only in Apps Script, never in HTML) ── var ADMIN_PASSWORD = "your-strong-password-here"; // ── Login handler — called by admin page to verify password ── function doGet(e) { var params = e.parameter; if (params.action === "login") { if (params.pwd === ADMIN_PASSWORD) { // Generate a simple time-based token var token = Utilities.base64Encode( ADMIN_PASSWORD + "|" + new Date().toDateString() ); return ContentService .createTextOutput(JSON.stringify({ status: "ok", token: token })) .setMimeType(ContentService.MimeType.JSON); } else { return ContentService .createTextOutput(JSON.stringify({ status: "fail" })) .setMimeType(ContentService.MimeType.JSON); } } return ContentService .createTextOutput(JSON.stringify({ status: "invalid" })) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { var data = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var imageUrls = []; // Process multiple base64 images if (data.images && data.images.length > 0) { var folderName = "Blog Images"; var folders = DriveApp.getFoldersByName(folderName); var folder = folders.hasNext() ? folders.next() : DriveApp.createFolder(folderName); for (var i = 0; i < data.images.length; i++) { var img = data.images[i]; var contentType = img.data.substring(img.data.indexOf(":") + 1, img.data.indexOf(";")); var base64Data = img.data.substring(img.data.indexOf(",") + 1); var decoded = Utilities.base64Decode(base64Data); var blob = Utilities.newBlob(decoded, contentType, img.name); var file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); // Construct web viewable Google Drive URL var fileUrl = "https://lh3.googleusercontent.com/d/" + file.getId(); imageUrls.push(fileUrl); } } var blogId = "B-" + Math.floor(Math.random() * 1000000) + "-" + new Date().getTime(); // Columns structure in sheet: // Timestamp | Blog ID | Title | Category | Author | Summary | Content | Image URLs (separated by |) sheet.appendRow([ new Date(), blogId, data.title || "", data.category || "", data.author || "", data.summary || "", data.content || "", imageUrls.join("|") // Pipe separated URLs ]); return ContentService .createTextOutput(JSON.stringify({ status: "success", id: blogId, urls: imageUrls })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: "error", message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
